@override
Future<num> getWidth() async {
  var params = <String, dynamic>{};
  params.putIfAbsent("userId", () => userId);

  return await methodChannel
      .invokeMethod<num>('getUserShareWidth', params)
      .then<num>((num? value) => value ?? 0);
}
